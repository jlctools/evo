23 #define EVO_IMPL_ARRAY_GROW_ADD(SIZE) { \

34 #define EVO_IMPL_ARRAY_GROW_INSERT(INDEX, SIZE) { \
35  assert( size_ > 0 ); \
36  Size newsize = size_ + SIZE; \
