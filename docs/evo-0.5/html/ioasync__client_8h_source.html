259  static void on_read(struct bufferevent* bev, void* self_handler) {
260  EVO_PARAM_UNUSED(bev);
261  String logstr;
262  ProtocolHandler& self = *(ProtocolHandler*)self_handler;
263  AsyncBuffers& bufs = self.bufs_;
264  bufs.attach_read();
265  if (self.read_fixed_size_ > 0) {
266  if (self.logger.check(LOG_LEVEL_DEBUG_LOW))
267  self.logger.log_direct(LOG_LEVEL_DEBUG_LOW, logstr.set().reserve(64) << "AsyncClient " << self.id_ << " fixed read: " << self.read_fixed_size_);
268  for (;;) {
269  SubString data;
270  if (!bufs.read_fixed(data, self.read_fixed_size_))
271  return; // wait for more data
272  self.read_fixed_size_ = 0;
273  if (!self.on_read_fixed(self.read_fixed_size_, data, NULL)) {
274  if (self.logger.check(LOG_LEVEL_DEBUG_LOW))
275  self.logger.log_direct(LOG_LEVEL_DEBUG_LOW, logstr.set().reserve(64) << "AsyncClient " << self.id_ << " on_read_fixed() returned false to close");
276  self.close();
277  return;
278  }
279  bufs.read_flush();
280  if (self.read_fixed_size_ <= 0)
281  break;
282  }
283  bufs.read_reset(ProtocolHandler::MIN_INITIAL_READ, self.max_read_size_);
284  if (bufs.read_size() == 0)
285  return;
286  }
287  if (self.logger.check(LOG_LEVEL_DEBUG_LOW))
288  self.logger.log_direct(LOG_LEVEL_DEBUG_LOW, logstr.set().reserve(64) << "AsyncClient " << self.id_ << " read: " << bufs.read_size());
289  if (!self.on_read(self.read_fixed_size_, bufs, NULL)) {
290  if (self.logger.check(LOG_LEVEL_DEBUG_LOW))
291  self.logger.log_direct(LOG_LEVEL_DEBUG_LOW, logstr.set().reserve(64) << "AsyncClient " << self.id_ << " on_read() returned false to close");
292  self.close();
293  return;
294  }
295  }

297  static void on_event(struct bufferevent* bev, short events, void* self_ptr) {
298  EVO_PARAM_UNUSED(bev);
299  String logstr;
300  This& self = *(This*)self_ptr;
301  if (events & BEV_EVENT_CONNECTED && self.state_ == sCONNECTING) {
302  self.logger.log(LOG_LEVEL_DEBUG_LOW, logstr.set().reserve(34) << "AsyncClient " << self.id_ << " connected");
303  self.state_ = sCONNECTED;
304  ((ProtocolHandler&)self).on_connect();
305  if (self.on_connect_ != NULL)
306  self.on_connect_->on_connect();
307  } else {
308  AsyncError err;
309  if (events & BEV_EVENT_EOF)
310  err = aeCLOSED;
311  else if (events & BEV_EVENT_TIMEOUT)
312  err = aeTIMEOUT;
313  else if (events & BEV_EVENT_READING)
314  err = (self.state_ == sCONNECTING ? aeCONNECT : aeIO_READ);
315  else if (events & BEV_EVENT_WRITING)
316  err = (self.state_ == sCONNECTING ? aeCONNECT : aeIO_WRITE);
317  else if (events & BEV_EVENT_CONNECTED) {
318  self.logger.log(LOG_LEVEL_ERROR, logstr.set().reserve(64) << "AsyncClient " << self.id_ << " error: Unexpected 'connected' event");
319  err = aeIO;
320  } else
321  err = aeIO;
322  if (self.logger.check(LOG_LEVEL_ERROR)) {
323  const SubString errmsg(async_error_msg(err));
324  self.logger.log_direct(LOG_LEVEL_ERROR, logstr.set().reserve(42 + errmsg.size()) << "AsyncClient " << self.id_ << " error: " << errmsg << " (code: " << (int)err << ')');
325  }
326 
327  self.close();
328  ((ProtocolHandler&)self).on_error(err);
329  if (self.on_error_ != NULL)
330  self.on_error_->on_error(err);
331  }
