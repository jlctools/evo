66  #define EVO_TEMP_LOG_ERR(MSG, ...) { \
67  if (use_stderr) \
68  fprintf(stderr, "daemonize: " MSG "\n", __VA_ARGS__); \
69  else \
70  syslog(LOG_ERR, "fork() failed: %s", strerror(errno)); \
71  }
72 
73  bool use_stderr = (flags & DAEMONIZE_USE_STDERR);
74  if (!use_stderr)
75  openlog("daemonize", LOG_PID, LOG_DAEMON);
76 
77  bool result = false;
78  for (;;) {
79  // Fork to detach from parent
80  pid_t pid = fork();
81  if (pid == -1) {
82  EVO_TEMP_LOG_ERR("fork() failed: %s", strerror(errno));
83  break;
84  }
85  if (pid > 0)
86  _exit(0); // parent
87 
88  // New session
89  setsid();
90 
91  // Fork again to completely detatch from terminal
92  pid = fork();
93  if (pid == -1) {
94  EVO_TEMP_LOG_ERR("fork() failed: %s", strerror(errno));
95  break;
96  }
97  if (pid > 0)
98  ::_exit(0); // parent
99 
100  // Reset umask, cd root
101  umask(0);
102  if ((flags & DAEMONIZE_NO_CD) == 0 && chdir("/") != 0)
103  EVO_TEMP_LOG_ERR("chdir() on / failed: %s", strerror(errno));
104 
105  // Redirect std in/out/err to /dev/null
106  if (flags & DAEMONIZE_NO_REDIR) {
107  close(STDIN_FILENO);
108  close(STDOUT_FILENO);
109  if (!use_stderr)
110  close(STDERR_FILENO);
111  if (open("/dev/null", O_RDWR) != 0) {
112  EVO_TEMP_LOG_ERR("open() on /dev/null failed: %s", strerror(errno));
113  exit(1);
114  }
115  if (dup(STDIN_FILENO) == -1)
116  EVO_TEMP_LOG_ERR("dup() failed for stdout: %s", strerror(errno));
117  if (!use_stderr && dup(STDIN_FILENO) == -1)
118  EVO_TEMP_LOG_ERR("dup() failed for stderr: %s", strerror(errno));
119  }
120 
121  // Success
122  result = true;
123  break;
124  }
125 
126  if (!use_stderr)
127  closelog();
128  if (!result && (flags & DAEMONIZE_NO_EXIT) == 0)
129  exit(1);
130  return result;
