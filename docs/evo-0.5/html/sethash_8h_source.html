664  // Search "others" for key, using compare, set index (insertion point if not found)
665  const Value* search(Size& index, const Value& key, const THash& compare) const {
666  int cmp;
667  Size left = 0, right = others.size(), mid = 0;
668  const Value* items = others.data();
669  while (left < right) {
670  mid = left + ((right-left) / 2);
671  const Value* item = items + mid;
672  cmp = compare(key, *item);
673  if (cmp < 0) {
674  right = mid;
675  } else if (cmp == 0) {
676  index = mid;
677  return item;
678  } else
679  left = mid + 1;
680  }
681  index = left;
682  return NULL;
683  }

735  data_.sizemask = newsize - 1;
736  data_.threshold = (Size)(newsize * THRESHOLD);
